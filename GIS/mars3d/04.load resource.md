# 加载各类资源
## # 地形资源
- 配置文件配置资源
```js
// 加载地形资源
const config = {
  terrain: {
    url: 'http://192.168.7.18/terrain/sichuan_terrain',
    show: true,
  },
};
// 关闭地形
map.hasTerrain = false
// 开启地形
map.hasTerrain = true
```
## # 影像栅格资源
- 在线影像-天地图
```js
// 加载 3857 坐标系天地图影像
const config = {
  basemaps: [
    {
      name: '天地图影像(EPSG:3857)',
      icon: 'img/basemaps/tdt_img.png',
      type: 'tdt',
      layer: 'img_d',
      show: true,
    },
  ],
};
```
```js
// 以分组的方式加载天地图影像和注记，同时控制渲染两个图层
const config = {
  basemaps: [
    {
      name: '天地图影像(EPSG:4326)',
      icon: 'img/basemaps/tdt_img.png',
      type: 'group',
      show: true,
      layers: [
        {
          name: '底图',
          type: 'tdt',
          layer: 'img_d',
          crs: 'EPSG:4326',
        },
        {
          name: '注记',
          type: 'tdt',
          layer: 'img_z',
          crs: 'EPSG:4326',
        },
      ],
    },
  ],
};
```
